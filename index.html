Original C++ Implementation

This web application is based on the original C++ terminal application. Here's the core implementation and comparison:

🔧 C++ Student Class

class Student {
private:
    std::string name;
    int rollNumber;
    double marks;
    std::string course;
    std::string email;
    
public:
    // Constructor
    Student(std::string n, int r, double m, 
            std::string c, std::string e) 
        : name(n), rollNumber(r), marks(m), 
          course(c), email(e) {}
    
    // Getters
    std::string getName() const { return name; }
    int getRollNumber() const { return rollNumber; }
    double getMarks() const { return marks; }
    std::string getCourse() const { return course; }
    std::string getEmail() const { return email; }
    
    // Calculate Grade
    char calculateGrade() const {
        if (marks >= 90) return 'A';
        if (marks >= 80) return 'B';
        if (marks >= 70) return 'C';
        if (marks >= 60) return 'D';
        return 'F';
    }
    
    // Display function
    void display() const {
        std::cout << std::setw(15) << name 
                  << std::setw(10) << rollNumber 
                  << std::setw(8) << std::fixed 
                  << std::setprecision(2) << marks
                  << std::setw(15) << course
                  << std::setw(8) << calculateGrade() 
                  << std::endl;
    }
};

🌐 JavaScript Equivalent

class Student {
    constructor(name, rollNumber, marks, course, email) {
        this.id = this.generateId();
        this.name = name;
        this.rollNumber = rollNumber;
        this.marks = marks;
        this.course = course;
        this.email = email;
        this.grade = this.calculateGrade(marks);
        this.dateAdded = new Date().toLocaleDateString();
    }
    
    calculateGrade(marks) {
        if (marks >= 90) return 'A';
        if (marks >= 80) return 'B';
        if (marks >= 70) return 'C';
        if (marks >= 60) return 'D';
        return 'F';
    }
    
    generateId() {
        return Date.now().toString(36) + 
               Math.random().toString(36).substr(2);
    }
    
    display() {
        return `
            <tr>
                <td>${this.name}</td>
                <td>${this.rollNumber}</td>
                <td>${this.email}</td>
                <td>${this.course}</td>
                <td>${this.marks}</td>
                <td>${this.grade}</td>
            </tr>
        `;
    }
}
